01061       // CASE I: Requested size is exactly existing size
01062       if (s == size()) {

01064       }

01067       if((s % CHUNK_SIZE) != 0 || (size() == 0))

01069 
01070       // CASE II: Requested size is smaller than existing size
01071       if (s < size()) {
01072         _e = _b + s; 
01073         // TODO: Is it ok we don't destroy anything? I think so.
01074       }
01075 
01076       // CASE III: Requested size is greater than existing size but smaller than capacity
01077       else if (s < _l._idx) {
01078         _e = uninitialized_fill(_chunk_a, end(), begin() + s, v);
01079       }
01080 
01081       // CASE IV: Requested size is greater than existing capacity
01082       else {
01083         T** tmp = _table_p;
01084         // Allocate a new chunk table
01085         _table_p = _table_a.allocate(new_table_size);
01086         uninitialized_fill(_table_a, _table_p, _table_p + new_table_size, pointer());
01087 
01088         // Copy in the old chunk table
01089         std::copy(tmp, tmp + _table_size, _table_p);
01090 
01091         // Add necessary additional chunks and map them into the chunk table 
01092         size_type num_new_chunks = new_table_size - _table_size;
01093         for (size_type i = 0; i < num_new_chunks; ++ i) {
01094           pointer _chunk_p = _chunk_a.allocate(CHUNK_SIZE); 
01095           uninitialized_fill(_chunk_a, _chunk_p, _chunk_p + CHUNK_SIZE, v);

01102 
01103         _table_size = new_table_size;
01104         _l += CHUNK_SIZE * num_new_chunks;
01105         _e = _b + s;
01106       }
01107       assert(valid());
01108     }
